require('dotenv').config(); const express = require('express'); const { createClient } = require('@supabase/supabase-js'); const fs = require('fs'); const path = require('path'); const multer = require('multer'); const app = express(); const upload = multer({ storage: multer.memoryStorage() }); // Configuration Express app.use(express.json()); app.use(express.urlencoded({ extended: true })); app.use(express.static('public')); // --- CONNEXION SUPABASE --- const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY); // Système de Logs async function logAction(type, message, meta = {}) { await supabase.from('system_logs').insert([{ action_type: type, message: message, metadata: meta }]); } // --- MOTEUR DE RENDU --- function render(viewName, variables = {}) { const filePath = path.join(__dirname, 'views', viewName); if (!fs.existsSync(filePath)) return `Erreur : La vue ${viewName} est introuvable.`; let template = fs.readFileSync(filePath, 'utf8'); // ÉTAPE 1 : On injecte d'abord le HTML et CSS custom // Cela permet aux variables comme {{nom}} d'apparaître dans le template final if (variables.html_custom) { template = template.replace(new RegExp('{{html_custom}}', 'g'), variables.html_custom); } if (variables.css_custom) { template = template.replace(new RegExp('{{css_custom}}', 'g'), variables.css_custom); } // ÉTAPE 2 : On remplace TOUTES les variables (nom, logo, etc.) // y compris celles qui viennent d'être injectées Object.keys(variables).forEach(key => { const value = variables[key] !== undefined ? variables[key] : ''; const regex = new RegExp(`{{\\s*${key}\\s*}}`, 'g'); template = template.replace(regex, value); }); return template; } // --- ROUTES --- // 1. API DE CRÉATION app.post('/api/creer-commerce', upload.single('logo_file'), async (req, res) => { try { const body = req.body; let publicUrl = ""; if (req.file) { const fileName = `${Date.now()}-${body.slug}.${req.file.originalname.split('.').pop()}`; await supabase.storage.from('logos').upload(`uploads/${fileName}`, req.file.buffer, { contentType: req.file.mimetype }); publicUrl = supabase.storage.from('logos').getPublicUrl(`uploads/${fileName}`).data.publicUrl; } await supabase.from('business').insert([{ nom: body.nom, slug: body.slug, gestionnaire_prenom: body.g_prenom, gestionnaire_nom: body.g_nom, gestionnaire_email: body.g_email, gestionnaire_tel: body.g_tel, prix_mensuel: body.prix_mensuel, statut: body.statut, is_active: true, config_design: { couleur: body.couleur, logo_url: publicUrl, dash_html: body.dash_html, dash_css: body.dash_css, card_html: body.card_html, card_css: body.card_css } }]); await logAction('CREATE', `Nouveau commerce : ${body.nom}`); res.redirect('/admin/inventory'); } catch (err) { res.status(500).send(err.message); } }); // 5. `).join(''); res.send(render('admin-logs.html', { content: logsHtml })); }); app.get('/', (req, res) => res.redirect('/admin/inventory')); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`🚀 Serveur : http://localhost:${PORT}`));
